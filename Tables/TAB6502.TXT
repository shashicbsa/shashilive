    VAR
      Text000@1000 : TextConst 'ENU=Entries exist for item %1. The field %2 cannot be changed.;ENA=Entries exist for item %1. The field %2 cannot be changed.';
      Item@1002 : Record 27;
      ItemLedgEntry@1003 : Record 32;
      Text001@1001 : TextConst 'ENU=%1 is %2 for item %3. The field %4 cannot be changed.;ENA=%1 is %2 for item %3. The field %4 cannot be changed.';
      Text002@1005 : TextConst 'ENU=You cannot delete %1 %2 because it is used on one or more items.;ENA=You cannot delete %1 %2 because it is used on one or more items.';

    PROCEDURE TestSetSpecific@2(CurrentFieldname@1000 : Text[100]);
    BEGIN
      Item.RESET;
      Item.SETRANGE("Item Tracking Code",Code);
      IF Item.FIND('-') THEN
        REPEAT
          ItemLedgEntry.RESET;
          ItemLedgEntry.SETCURRENTKEY("Item No.");
          ItemLedgEntry.SETRANGE("Item No.",Item."No.");
          IF ItemLedgEntry.FINDFIRST THEN
            ERROR(
              Text000,
              Item."No.",CurrentFieldname);
        UNTIL Item.NEXT = 0;
    END;

    PROCEDURE TestRemoveSpecific@3(CurrentFieldname@1000 : Text[100]);
    BEGIN
      Item.RESET;
      Item.SETRANGE("Item Tracking Code",Code);
      IF Item.FIND('-') THEN
        REPEAT
          IF Item."Costing Method" = Item."Costing Method"::Specific THEN
            ERROR(
              Text001,
              Item.FIELDCAPTION("Costing Method"),
              Item."Costing Method",Item."No.",CurrentFieldname);
          ItemLedgEntry.RESET;
          ItemLedgEntry.SETCURRENTKEY("Item No.");
          ItemLedgEntry.SETRANGE("Item No.",Item."No.");
          IF ItemLedgEntry.FINDFIRST THEN
            ERROR(
              Text000,
              Item."No.",CurrentFieldname);
        UNTIL Item.NEXT = 0;
    END;

    PROCEDURE TestDelete@1();
    BEGIN
      Item.RESET;
      Item.SETRANGE("Item Tracking Code",Code);
      IF NOT Item.ISEMPTY THEN
        ERROR(Text002,TABLECAPTION,Code);
    END;

    BEGIN
    END.
